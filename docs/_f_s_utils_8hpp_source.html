129 size_t FSUtils::lineCount(const std::string &fileName){
130  const char *fname = fileName.c_str();
131  static const auto BUF_SIZE = 16*1024;
132  int fd = open(fname, O_RDONLY);
133  if(fd == -1)
134  throw std::runtime_error("Couldn't open " + std::string(fname));
135 
136  char buf[BUF_SIZE + 1];
137  size_t lines = 0;
138 
139  while(size_t bytes_read = read(fd, buf, BUF_SIZE)){
140  if(bytes_read == (size_t) -1)
141  throw std::runtime_error("Read Failure" + std::string(fname));
142 
143  if(!bytes_read)
144  break;
145 
146  for(char *p = buf; (p = (char *) memchr(p, '\n', (buf + bytes_read) - p)); ++p)
147  ++lines;
148  }
149  close(fd);
150  return lines;
151 }

160 std::vector<std::string> FSUtils::getFilesInDir(const std::string &dirPath, const std::string &ext){
161  std::vector<std::string> files;
162 
163  DIR *dir;
164  struct dirent *ent;
165 
166  if((dir = opendir(dirPath.c_str())) == NULL)
167  throw std::runtime_error("Couldn't open " + dirPath);
168 
169  // print all the files and directories within directory
170  while ((ent = readdir(dir)) != NULL) {
171  if(isFile(std::string(dirPath + "/" + ent->d_name))){
172  // Only add files with the requested file extension
173  if(ext != ""){
174  // Note: This can easily be fooled
175  std::string f_name(ent->d_name);
176  if(StrUtils::endsWithString(f_name, ext))
177  files.push_back(f_name);
178  }
179  // Don't care what the file extension is
180  else
181  files.emplace_back(ent->d_name);
182  }
183  }
184  closedir(dir);
185 
186  return files;
187 }

341 bool FSUtils::deleteDir(const std::string &dirpath){
342  if(!dexists(dirpath))
343  return true;
344 
345  DIR *dir;
346  struct dirent *ent;
347 
348  if((dir = opendir(dirpath.c_str())) == NULL)
349  throw std::runtime_error("Couldn't open " + dirpath + " for deletion");
350 
351  while((ent = readdir(dir)) != NULL){
352  std::string it(ent->d_name);
353 
354  // Be very careful not to traverse vertically out of the directory
355  if(THIS_DIR_DOT == it || PREV_DIR_DOT == it)
356  continue;
357 
358  std::string relative_it(dirpath + "/" + it);
359 
360  if(isDir(relative_it)){
361  // This really shouldn't ever be the case, since we couldn't be in this loop
362  if(dempty(relative_it))
363  std::remove(relative_it.c_str());
364  // Make the recursive call
365  else
366  deleteDir(relative_it);
367  }
368  // Remove all the files from the directory
369  else if(isFile(relative_it))
370  std::remove(relative_it.c_str());
371  else
372  throw std::runtime_error("Is that a symlink?? Uh...not implemented!");
373  }
374  closedir(dir);
375 
376  if(isDir(dirpath) && dempty(dirpath))
377  std::remove(dirpath.c_str());
378 
379  return !dexists(dirpath);
380 }

518 float FSUtils::fsize(const std::string &filepath, const std::string &order){
519  if(!fexists(filepath))
520  throw std::runtime_error("Cannot find " + filepath);
521  float sz;
522 
523  {
524  std::ifstream f(filepath, std::ifstream::ate | std::ifstream::binary);
525  sz = static_cast<float>(f.tellg());
526  }
527 
528  if("b" == order)
529  return sz;
530  else if("kb" == order)
531  return sz / 1000;
532  else if("mb" == order)
533  return sz / 1000000;
534  else if("gb" == order)
535  return sz / 1000000000;
536  else
537  return sz;

591 std::string FSUtils::getPermissions(const std::string &path){
592  if(!dexists(path) && !fexists(path))
593  throw std::runtime_error("Can't find " + path);
594 
595  struct stat buffer;
596  stat(path.c_str(), &buffer);
597  auto user = (buffer.st_mode & S_IRWXU) >> 6;
598  auto group = (buffer.st_mode & S_IRWXG) >> 3;
599  auto world = (buffer.st_mode & S_IRWXO);

602  return perms;
