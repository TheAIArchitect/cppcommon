123 size_t FSUtils::lineCount(const std::string &fileName){
124  const char *fname = fileName.c_str();
125  static const auto BUF_SIZE = 16*1024;
126  int fd = open(fname, O_RDONLY);
127  if(fd == -1)
128  throw std::runtime_error("Couldn't open " + std::string(fname));
129 
130  char buf[BUF_SIZE + 1];
131  size_t lines = 0;
132 
133  while(size_t bytes_read = read(fd, buf, BUF_SIZE)){
134  if(bytes_read == (size_t) -1)
135  throw std::runtime_error("Read Failure" + std::string(fname));
136 
137  if(!bytes_read)
138  break;
139 
140  for(char *p = buf; (p = (char *) memchr(p, '\n', (buf + bytes_read) - p)); ++p)
141  ++lines;
142  }
143  close(fd);
144  return lines;
145 }

154 std::vector<std::string> FSUtils::getFilesInDir(const std::string &dirPath, const std::string &ext){
155  std::vector<std::string> files;
156 
157  DIR *dir;
158  struct dirent *ent;
159 
160  if((dir = opendir(dirPath.c_str())) == NULL)
161  throw std::runtime_error("Couldn't open " + dirPath);
162 
163  // print all the files and directories within directory
164  while ((ent = readdir(dir)) != NULL) {
165  if(isFile(std::string(dirPath + "/" + ent->d_name))){
166  // Only add files with the requested file extension
167  if(ext != ""){
168  // Note: This can easily be fooled
169  std::string f_name(ent->d_name);
170  if(StrUtils::endsWithString(f_name, ext))
171  files.push_back(f_name);
172  }
173  // Don't care what the file extension is
174  else
175  files.emplace_back(ent->d_name);
176  }
177  }
178  closedir(dir);
179 
180  return files;
181 }

332 bool FSUtils::deleteDir(const std::string &dirpath){
333  if(!dexists(dirpath))
334  return true;
335 
336  DIR *dir;
337  struct dirent *ent;
338 
339  if((dir = opendir(dirpath.c_str())) == NULL)
340  throw std::runtime_error("Couldn't open " + dirpath + " for deletion");
341 
342  while((ent = readdir(dir)) != NULL){
343  std::string it(ent->d_name);
344 
345  // Be very careful not to traverse vertically out of the directory
346  if(THIS_DIR_DOT == it || PREV_DIR_DOT == it)
347  continue;
348 
349  std::string relative_it(dirpath + "/" + it);
350 
351  if(isDir(relative_it)){
352  // This really shouldn't ever be the case, since we couldn't be in this loop
353  if(dempty(relative_it))
354  std::remove(relative_it.c_str());
355  // Make the recursive call
356  else
357  deleteDir(relative_it);
358  }
359  // Remove all the files from the directory
360  else if(isFile(relative_it))
361  std::remove(relative_it.c_str());
362  else
363  throw std::runtime_error("Not sure what else we could have here...");
364  }
365  closedir(dir);
366 
367  if(isDir(dirpath) && dempty(dirpath))
368  std::remove(dirpath.c_str());
369 
370  return !dexists(dirpath);
371 }

505 float FSUtils::fsize(const std::string &filepath, const std::string &order){

508  float sz;
509

576 std::string FSUtils::getPermissions(const std::string &path){
577  if(!dexists(path) && !fexists(path))
578  throw std::runtime_error("Can't find " + path);
579 
580  struct stat buffer;
581  stat(path.c_str(), &buffer);
582  auto user = (buffer.st_mode & S_IRWXU) >> 6;
583  auto group = (buffer.st_mode & S_IRWXG) >> 3;
584  auto world = (buffer.st_mode & S_IRWXO);
585 
586  auto perms = std::to_string(user) + std::to_string(group) + std::to_string(world);
587  return perms;
