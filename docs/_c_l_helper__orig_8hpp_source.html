155  device_id = device_ids[deviceToUse];
156  print_selected_device_info(device_id, deviceToUse);
157 
158  context = clCreateContext(0, 1, &device_id, nullptr, nullptr, &err);
159  check_and_print_cl_err(err);
160

162  check_and_print_cl_err(err);
163  return true;

171  void read_kernels_from_file(const char *fname, char **programBuffer){
173  /* FROM:
174  -> http://dhruba.name/2012/08/16/opencl-cookbook-creating-programs-and-reading-kernels-from-a-file/
175  */
176  // get size of kernel source
177  FILE *programHandle = fopen(fname, "r");
178  fseek(programHandle, 0, SEEK_END);
179  int programSize = (int)ftell(programHandle);
180  rewind(programHandle);
181 
182  // read kernel source into buffer
183  (*programBuffer) = (char*) malloc(programSize + 1);
184  (*programBuffer)[programSize] = '\0';
185  fread((*programBuffer), sizeof(char), programSize, programHandle);
186  fclose(programHandle);
187  }

222  void print_selected_device_info(cl_device_id cdi, int device_num){
223 
224  char* value;
225  size_t valueSize;
226  cl_uint maxComputeUnits;
227 
228  // print device name
229  clGetDeviceInfo(cdi, CL_DEVICE_NAME, 0, NULL, &valueSize);
230  value = (char*) malloc(valueSize);
231  clGetDeviceInfo(cdi, CL_DEVICE_NAME, valueSize, value, NULL);
232  printf("%d. Device: %s\n", device_num, value);
233  free(value);
234 
235  // print hardware device version
236  clGetDeviceInfo(cdi, CL_DEVICE_VERSION, 0, NULL, &valueSize);
237  value = (char*) malloc(valueSize);
238  clGetDeviceInfo(cdi, CL_DEVICE_VERSION, valueSize, value, NULL);
239  printf(" %d.%d Hardware version: %s\n", device_num, 1, value);
240  free(value);
241 
242  // print software driver version
243  clGetDeviceInfo(cdi, CL_DRIVER_VERSION, 0, NULL, &valueSize);
244  value = (char*) malloc(valueSize);
245  clGetDeviceInfo(cdi, CL_DRIVER_VERSION, valueSize, value, NULL);
246  printf(" %d.%d Software version: %s\n", device_num, 2, value);
247  free(value);
248 
249  // print c version supported by compiler for device
250  clGetDeviceInfo(cdi, CL_DEVICE_OPENCL_C_VERSION, 0, NULL, &valueSize);
251  value = (char*) malloc(valueSize);
252  clGetDeviceInfo(cdi, CL_DEVICE_OPENCL_C_VERSION, valueSize, value, NULL);
253  printf(" %d.%d OpenCL C version: %s\n", device_num, 3, value);
254  free(value);
255 
256  // print parallel compute units
257  clGetDeviceInfo(cdi, CL_DEVICE_MAX_COMPUTE_UNITS, sizeof(maxComputeUnits), &maxComputeUnits, NULL);
258  printf(" %d.%d Parallel compute units: %d\n", device_num, 4, maxComputeUnits);
259  }
