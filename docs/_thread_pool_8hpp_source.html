21 class ThreadPool {
22 public:
23  ThreadPool(size_t);
24  template<class F, class... Args>
25  auto push(F&& f, Args&&... args) -> std::future<typename std::result_of<F(Args...)>::type>;
26  ~ThreadPool();
27 private:
28  // need to keep track of threads so we can join them
29  std::vector<std::thread> mWorkers;
30  // the task queue
31  std::queue< std::function<void()>> mTasks;
32 
33  // synchronization
34  std::mutex mQueueMutex;
35  std::condition_variable mCondVar;
36  bool mStop;
37 };

39 // the constructor just launches some amount of workers
40 inline ThreadPool::ThreadPool(size_t threads)
41  : mStop(false)
42 {
43  for(size_t i = 0;i<threads;++i){
44  mWorkers.emplace_back(
45  [this]{
46  while(true){
47  std::function<void()> task;
48 
49  {
50  std::unique_lock<std::mutex> lock(this->mQueueMutex);
51  this->mCondVar.wait(lock,
52  [this]{ return this->mStop || !this->mTasks.empty(); });
53  if(this->mStop && this->mTasks.empty())
54  return;
55  task = std::move(this->mTasks.front());
56  this->mTasks.pop();
57  }
58  task();
59  }
60  });
61  }
62 }

64 // add new work item to the pool
65 template<class F, class... Args>
66 auto ThreadPool::push(F&& f, Args&&... args) -> std::future<typename std::result_of<F(Args...)>::type>{
67  using return_type = typename std::result_of<F(Args...)>::type;
68 
69  auto task = std::make_shared< std::packaged_task<return_type()>>(
70  std::bind(std::forward<F>(f), std::forward<Args>(args)...)
71  );
72 
73  std::future<return_type> res = task->get_future();
74  {
75  std::unique_lock<std::mutex> lock(mQueueMutex);
76 
77  // don't allow enqueueing after stopping the pool
78  if(mStop)
79  throw std::runtime_error("push on stopped ThreadPool");
80 
81  mTasks.emplace([task](){ (*task)(); });
82  }
83  mCondVar.notify_one();
84  return res;
85 }

88 inline ThreadPool::~ThreadPool()
89 {
90  {
91  std::unique_lock<std::mutex> lock(mQueueMutex);
92  mStop = true;
93  }
94  mCondVar.notify_all();
95  for(std::thread &worker: mWorkers)
96  worker.join();
97 }
