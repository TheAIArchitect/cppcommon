43  // Removes leading and trailing whitespace from a string
44  inline std::string removeLTWS(std::string &trimmed){
45  if(trimmed.empty()) return "";
46  // Create a new string without the leading whitespace
47  auto first_char = trimmed.find_first_not_of(" ");
48  auto last_char = trimmed.size() - first_char;
49  std::string no_leading_ws{trimmed.substr(first_char, last_char)};
50 
51  // Reset first char to 0;
52  first_char = 0;
53 
54  // Create a new string without the trailing whitespace
55  last_char = no_leading_ws.find_last_not_of(" ");
56  return no_leading_ws.substr(first_char, last_char + 1);
57  }
58 }

65 std::string StrUtils::removeCharFromStr(const std::string &s, const char ch){

68  if(s.length() == 0)
69  return "";

73  return s;

77  std::string test_s = s.substr(0, 1);
78  std::string rest = s.substr(1, end);

126 std::string StrUtils::trimSpecific(const std::string &s, const std::string &nonPrintCharsToRemove, bool removeWS){
127  if((s.empty() || s.find_first_not_of(nonPrintCharsToRemove) == std::string::npos) && (!nonPrintCharsToRemove.empty()))
128  return "";
129 
130  std::string trimmed{s};
131 
132  // Loop through the supplied characters and clean it up
133  for(auto &&ch : nonPrintCharsToRemove)
134  trimmed.erase(remove(std::begin(trimmed), std::end(trimmed), ch), std::end(trimmed));
135 
136  return removeWS ? removeLTWS(trimmed) : trimmed;
137 }
