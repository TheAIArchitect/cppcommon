56 std::string Utils::timeStamp() {
57  using clock_t = std::chrono::system_clock;
58  auto now = clock_t::now();
59  auto count = clock_t::to_time_t(now);
60  auto secs = std::chrono::time_point_cast<std::chrono::milliseconds>(now);
61  auto partial = now - secs;
62  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(partial);
63 
64  std::ostringstream oss;
65  oss << std::put_time(std::localtime(&count), "%Y-%m-%d %X");
66  oss << "." << ms.count();
67  return oss.str();
68 }
