63 std::string FileUtils::readFullFile(const std::string &fileName){
64  std::ifstream ifs(fileName, std::ios::in | std::ios::binary | std::ios::ate);
65  if(!ifs.is_open())
66  throw std::runtime_error("Couldn't open " + fileName);
67 
68  std::ifstream::pos_type f_size = ifs.tellg();
69  ifs.seekg(0, std::ios::beg);
70 
71  std::vector<char> bytes(f_size);
72  ifs.read(bytes.data(), f_size);
73 
74  return std::string(bytes.data(), f_size);

106 size_t FileUtils::lineCount(const std::string &fileName){
107  const char *fname = fileName.c_str();
108  static const auto BUF_SIZE = 16*1024;
109  int fd = open(fname, O_RDONLY);
110  if(fd == -1)
111  throw std::runtime_error("Couldn't open " + std::string(fname));
112 
113  char buf[BUF_SIZE + 1];
114  size_t lines = 0;
115 
116  while(size_t bytes_read = read(fd, buf, BUF_SIZE)){
117  if(bytes_read == (size_t) -1)
118  throw std::runtime_error("Read Failure" + std::string(fname));
119 
120  if(!bytes_read)
121  break;
122 
123  for(char *p = buf; (p = (char *) memchr(p, '\n', (buf + bytes_read) - p)); ++p)
124  ++lines;
125  }
126  close(fd);
127  return lines;
128 }

136 std::vector<std::string> FileUtils::getFilesInDir(const std::string &dirPath, const std::string &ext){
137  std::vector<std::string> files;
138 
139  DIR *dir;
140  struct dirent *ent;
141 
142  if((dir = opendir(dirPath.c_str())) == NULL)
143  throw std::runtime_error("Couldn't open " + dirPath);
144 
145  // print all the files and directories within directory
146  while ((ent = readdir(dir)) != NULL) {
147  if(isFile(std::string(dirPath + "/" + ent->d_name))){
148  // Only add files with the requested file extension
149  if(ext != ""){
150  // Note: This can easily be fooled
151  std::string f_name(ent->d_name);
152  if(StrUtils::endsWithString(f_name, ext))
153  files.push_back(f_name);
154  }
155  // Don't care what the file extension is
156  else
157  files.emplace_back(ent->d_name);
158  }
159  }
160  closedir(dir);
161 
162  return files;
163 }

311 bool FileUtils::deleteDir(const std::string &dirpath){
312  if(!dexists(dirpath))
313  return true;
314 
315  DIR *dir;
316  struct dirent *ent;
317 
318  if((dir = opendir(dirpath.c_str())) == NULL)
319  throw std::runtime_error("Couldn't open " + dirpath + " for deletion");
320 
321  while((ent = readdir(dir)) != NULL){
322  std::string it(ent->d_name);
323 
324  // Be very careful not to traverse vertically out of the directory
325  if(THIS_DIR_DOT == it || PREV_DIR_DOT == it)
326  continue;
327 
328  std::string relative_it(dirpath + "/" + it);
329 
330  if(isDir(relative_it)){
331  // This really shouldn't ever be the case, since we couldn't be in this loop
332  if(dempty(relative_it))
333  std::remove(relative_it.c_str());
334  // Make the recursive call
335  else
336  deleteDir(relative_it);
337  }
338  // Remove all the files from the directory
339  else if(isFile(relative_it))
340  std::remove(relative_it.c_str());
341  else
342  throw std::runtime_error("Not sure what else we could have here...");
343  }
344  closedir(dir);
345 
346  if(isDir(dirpath) && dempty(dirpath))
347  std::remove(dirpath.c_str());
348 
349  return !dexists(dirpath);
350 }

484  double sz;
