62 T NumUtils::getMedian(std::vector<T> &vec){
63  size_t v_size = vec.size();
64  // will automatically round down due to int division, subtract 1 due to 0 indexing
65  size_t mid = (v_size/2) - 1;
66  size_t midpo = mid + 1;
67  if (v_size % 2 == 1){
68  std::nth_element(vec.begin(), vec.begin()+(midpo), vec.end(),
69  [](T const a, T const b){
70  return b > a;
71  });
72  return vec[midpo];
73  }
74  else{
75  std::nth_element(vec.begin(), vec.begin()+(mid), vec.end(),
76  [](T const a, T const b){
77  return b > a;
78  });
79  T m1 = vec[mid];
80  std::nth_element(vec.begin(), vec.begin()+(midpo), vec.end(),
81  [](T const a, T const b){
82  return b > a;
83  });
84  T m2 = vec[midpo];
85  return ((m1+m2)/2.f);
86 
87  }
