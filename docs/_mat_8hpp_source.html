365  int from = static_cast<int>((slice * lhs.rows()) / num_threads);
366  int to = static_cast<int>(((slice + 1) * lhs.rows()) / num_threads);
367  auto rhs_c = rhs.cols();

398 inline Mat operator*(const Mat &lhs, const Mat &rhs) {
399  if(lhs.cols() != rhs.rows())
400  throw std::runtime_error("lhs cols != rhs rows");
401 
402  // If it's tiny, just multiply it out and move on
403  if(rhs.size() < DOT_TRANSPOSE)
404  return dot_(lhs, rhs);
405  // If it's medium size, take the time to transpose rhs and then multiply it out
406  else if(lhs.size() < DOT_THREADED && rhs.size() < DOT_THREADED)
407  return dot_transpose_(lhs, rhs);
408  // If the matrices are large, take the time to spin up some threads and multiply everything out
409  else
410  return dot_mt_(lhs, rhs);
411 }
