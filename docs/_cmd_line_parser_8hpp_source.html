37  CmdLineParser(int argc, const char **argv){
38  // If argc == 2 then there's just a single flag, no variable associated with it
39  // Place the flag in the map and move on
40  if(argc == 2){
41  std::string arg{argv[1]};
42  // It's got the leading dash
43  if(arg[0] == '-'){
44  args_map_.insert(std::make_pair(arg, ""));
45  args_map_.insert(std::make_pair(arg.substr(1,1), ""));
46  }
47  // No leading dash, add it anyway since it's a single flag-only argment list
48  else{
49  args_map_.insert(std::make_pair(arg, ""));
50  args_map_.insert(std::make_pair("-"+arg, ""));
51  }
52  // Skip the rest of the function
53  return;
54  }
55 
56  // Start at 1 to avoid the program name
57  for(auto i = 1; i < argc; ++i){
58  std::string arg{argv[i]}; // The raw argument
59  std::string flag; // The flag with the leading dash removed
60  std::string flag_char; // The flag with the leading dash attached
61  std::string var; // The variable associated with a flag
62  // Extra processing to deal with -wworld type flags where the variable and flag are not
63  // separated by whitespace
64  if(arg[0] == '-'){
65  // Split the flag from the variable
66  if(arg.length() > 2){
67  flag = arg.substr(1,1);
68  flag_char = arg.substr(0,2);
69  var = arg.substr(2);
70 
71  // Insert both lookups and the variable into the map and continue the loop
72  args_map_.insert(std::make_pair(flag, var));
73  args_map_.insert(std::make_pair(flag_char, var));
74  }
75  // Flag and variable are not stuck together, just process
76  else{
77  flag = arg.substr(1,1);
78  // Incremenet counter (i) to grab the next element from the argv char array
79  // Make sure to check that we're still in the proper range of argc
80  // If we're at the end, set var to "" to handle a flag being the last element in argv
81  if(++i < argc) {
82  var = argv[i];
83  // Undo the increment of i if the next element is another flag, let the loop counter increment
84  if(var[0] == '-') --i;
85  }
86  else
87  var = "";
88 
89  // Check if we have another flag or if it's a variable associated with the current flag
90  if(var[0] == '-' || var == ""){
91  args_map_.insert(std::make_pair(flag, ""));
92  args_map_.insert(std::make_pair(arg, ""));
93  }
94  else {
95  // arg already represents flag_char, no need to make a copy and we know we don't have another
96  // flag so add it as a variable
97  args_map_.insert(std::make_pair(flag, var));
98  args_map_.insert(std::make_pair(arg, var));
99  }
100  }
101  }
102  }
103  }
